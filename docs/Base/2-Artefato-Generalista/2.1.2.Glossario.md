# Glossário

## Introdução

Um glossário é uma ferramenta importante em diferentes áreas do conhecimento, pois contribui para tornar a compreensão e a comunicação mais claras e eficientes. Trata-se de uma lista de termos acompanhados de suas respectivas definições, organizada de forma alfabética ou por temas. Seu principal objetivo é explicar conceitos, expressões técnicas ou palavras específicas que podem não ser familiares ou que geram dúvidas para quem está começando ou se aprofundando em determinado assunto.

<script>
    const terms = [
        { id: 1, title: "Carpooling", desc: "Prática de compartilhar viagens de carro com outras pessoas que têm destinos semelhantes." },
        { id: 2, title: "Ponto de Encontro", desc: "Local pré-definido no campus ou cidade para o embarque e desembarque seguro dos passageiros." },
        { id: 3, title: "Match de Rota", desc: "Algoritmo que conecta motoristas e passageiros com trajetos e horários compatíveis." },
        { id: 4, title: "Vaga Solidária", desc: "Assento disponível oferecido pelo motorista sem fins lucrativos, apenas para divisão de custos." },
        { id: 5, title: "Comunidade FCTE", desc: "Grupo restrito de usuários verificados pertencentes à faculdade, garantindo maior segurança." },
        { id: 6, title: "Check-in", desc: "Confirmação de que o passageiro entrou no veículo, ativando o monitoramento da viagem." },
        { id: 7, title: "Rateio", desc: "Divisão proporcional dos custos de combustível e pedágio entre os ocupantes do veículo." },
        { id: 8, title: "Perfil Verificado", desc: "Status de segurança atribuído a usuários que validaram seus documentos e vínculo acadêmico." }
    ];

    const container = document.getElementById('cardsContainer');
    const dotsContainer = document.getElementById('dotsContainer');
    const modal = document.getElementById('modalOverlay');
    const modalTitle = document.getElementById('modalTitle');
    const modalBody = document.getElementById('modalBody');
    const closeBtn = document.getElementById('closeBtn');

    let currentIndex = 0;
    let startX = 0;
    let currentTranslate = 0;
    let prevTranslate = 0;
    let isDragging = false;

    // Initialize Cards
    function init() {
        terms.forEach((term, index) => {
            // Create Card
            const card = document.createElement('div');
            card.className = `card ${index === 0 ? 'active' : ''}`;
            card.innerHTML = `
                <div class="card-number">${term.id}</div>
                <h2 class="card-title">${term.title}</h2>
                <p class="card-description">${term.desc.substring(0, 80)}...</p>
            `;
            card.onclick = () => openModal(term);
            container.appendChild(card);

            // Create Dot
            const dot = document.createElement('div');
            dot.className = `dot ${index === 0 ? 'active' : ''}`;
            dot.onclick = () => goToSlide(index);
            dotsContainer.appendChild(dot);
        });
        updateSlider();
    }

    function openModal(term) {
        modalTitle.innerText = term.title;
        modalBody.innerText = term.desc;
        modal.style.display = 'flex';
    }

    closeBtn.onclick = () => modal.style.display = 'none';
    window.onclick = (e) => { if(e.target == modal) modal.style.display = 'none'; };

    // Slider Logic
    const sliderWrapper = document.getElementById('sliderWrapper');

    sliderWrapper.addEventListener('mousedown', dragStart);
    sliderWrapper.addEventListener('touchstart', dragStart);
    sliderWrapper.addEventListener('mouseup', dragEnd);
    sliderWrapper.addEventListener('touchend', dragEnd);
    sliderWrapper.addEventListener('mousemove', dragAction);
    sliderWrapper.addEventListener('touchmove', dragAction);

    function dragStart(e) {
        isDragging = true;
        startX = getPositionX(e);
        container.style.transition = 'none';
    }

    function dragAction(e) {
        if (!isDragging) return;
        const currentX = getPositionX(e);
        const diff = currentX - startX;
        currentTranslate = prevTranslate + diff;
        container.style.transform = `translateX(${currentTranslate}px)`;
    }

    function dragEnd() {
        isDragging = false;
        const movedBy = currentTranslate - prevTranslate;

        if (movedBy < -50 && currentIndex < terms.length - 1) currentIndex++;
        if (movedBy > 50 && currentIndex > 0) currentIndex--;

        goToSlide(currentIndex);
    }

    function getPositionX(e) {
        return e.type.includes('mouse') ? e.pageX : e.touches[0].clientX;
    }

    function goToSlide(index) {
        currentIndex = index;
        const cardWidth = 310; // card min-width + margin
        currentTranslate = -currentIndex * cardWidth;
        prevTranslate = currentTranslate;
        updateSlider();
    }

    function updateSlider() {
        container.style.transition = 'transform 0.5s cubic-bezier(0.25, 1, 0.5, 1)';
        container.style.transform = `translateX(${currentTranslate}px)`;
        
        // Update active classes
        document.querySelectorAll('.card').forEach((c, i) => {
            c.classList.toggle('active', i === currentIndex);
        });
        document.querySelectorAll('.dot').forEach((d, i) => {
            d.classList.toggle('active', i === currentIndex);
        });
    }

    init();
</script>
